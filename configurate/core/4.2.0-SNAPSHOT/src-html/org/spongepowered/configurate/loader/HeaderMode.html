017package org.spongepowered.configurate.loader;
018
019/**
020 * Modes which {@link ConfigurationLoader}s can use to manipulate headers
021 * when loading/saving.
022 *
023 * @since 4.0.0
024 */
025public enum HeaderMode {
026
027    /**
028     * Use the header loaded from an existing file, replacing any header set in
029     * the options.
030     */
031    PRESERVE,
032
033    /**
034     * Ignore any header present in input, and output a header if one has been
035     * set in the options.
036     */
037    PRESET,
038
039    /**
040     * Ignore any header present in input, and do not output any header.
041     */
042    NONE
043
044}
