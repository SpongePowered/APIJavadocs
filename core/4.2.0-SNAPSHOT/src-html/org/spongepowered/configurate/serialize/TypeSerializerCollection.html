095    /**
096     * Resolve a type serializer.
097     *
098     * <p>First, all registered serializers from this collection are queried in
099     * registration order, then if a parent collection is set, that collection
100     * is queried.</p>
101     *
102     * @param token the type a serializer is required for
103     * @param <T> the type to serialize
104     * @return a serializer if any is present, or null if no applicable
105     *          serializer is found
106     * @since 4.0.0
107     */
108    @SuppressWarnings("unchecked")
109    public <T> @Nullable TypeSerializer<T> get(final TypeToken<T> token) {
110        requireNonNull(token, "type");
111        return (TypeSerializer<T>) this.get0(token.getAnnotatedType());
112    }

114    /**
115     * Resolve a type serializer.
116     *
117     * <p>First, all registered serializers from this collection are queried in
118     * registration order, then if a parent collection is set, that collection
119     * is queried.</p>
120     *
121     * <p>This method will fail when provided a raw parameterized type</p>
122     *
123     * @param token the type a serializer is required for
124     * @param <T> the type to serialize
125     * @return a serializer if any is present, or null if no applicable
126     *          serializer is found
127     * @since 4.0.0
128     */
129    @SuppressWarnings("unchecked")
130    public <T> @Nullable TypeSerializer<T> get(final Class<T> token) {
131        requireNonNull(token, "type");
132        requireCompleteParameters(token);
133
134        return (TypeSerializer<T>) this.get((Type) token);
135    }

137    /**
138     * Resolve a type serializer.
139     *
140     * <p>First, all registered serializers from this collection are queried
141     * then if a parent collection is set, that collection is queried.

152    /**
153     * Resolve a type serializer with annotation information.
154     *
155     * <p>First, all registered serializers from this collection are queried
156     * then if a parent collection is set, that collection is queried.</p>
157     *
158     * <p>The returned serializer may not necessarily use the provided
159     * annotation information.</p>
160     *
161     * @param type the type a serializer is required for
162     * @return a serializer if any is present, or null if no applicable
163     *          serializer is found
164     * @since 4.2.0
165     */
166    public @Nullable TypeSerializer<?> get(final AnnotatedType type) {
167        return this.get0(GenericTypeReflector.toCanonicalBoxed(type));
168    }

170    private @Nullable TypeSerializer<?> get0(final AnnotatedType canonical) {
171        @Nullable TypeSerializer<?> serial = this.annotatedTypeMatches.computeIfAbsent(canonical, param -> {
172            for (final RegisteredSerializer ent : this.serializers) {
173                if (ent.matches(param)) {
174                    return ent.serializer();
175                }
176            }
177            return NoOp.INSTANCE;
178        });
179
180        if (serial == NoOp.INSTANCE) {
181            serial = null;
182        }
183
184        if (serial == null && this.parent != null) {
185            serial = this.parent.get0(canonical);
186        }
187        return serial;
188    }
189
190    private @Nullable TypeSerializer<?> get0(final Type canonical) {
191        @Nullable TypeSerializer<?> serial = this.typeMatches.computeIfAbsent(canonical, param -> {
192            for (final RegisteredSerializer ent : this.serializers) {
193                if (ent.matches(param)) {
194                    return ent.serializer();
195                }
196            }
197            return NoOp.INSTANCE;
198        });
199
200        if (serial == NoOp.INSTANCE) {
201            serial = null;
202        }
203
204        if (serial == null && this.parent != null) {
205            serial = this.parent.get0(canonical);
206        }
207        return serial;
208    }

229    @Override public boolean equals(final Object other) {
230        if (this == other) {
231            return true;
232        }
233        if (!(other instanceof TypeSerializerCollection)) {
234            return false;
235        }
236        final TypeSerializerCollection that = (TypeSerializerCollection) other;
237        return Objects.equals(this.parent, that.parent)
238                && this.serializers.equals(that.serializers);
239    }

243        return Objects.hash(this.parent, this.serializers);

272    /**
273     * A builder to construct new serializer collections.
274     *
275     * <p>Serializers added to a builder will be prioritized based on
276     * registration order, so if multiple serializers could match a single type,
277     * the first-registered one will be used.</p>
278     *
279     * @since 4.0.0
280     */
281    public static class Builder {
282        private final @Nullable TypeSerializerCollection parent;
283        private final List<RegisteredSerializer> serializers = new ArrayList<>();
284
285        Builder(final @Nullable TypeSerializerCollection parent) {
286            this.parent = parent;
287        }
288
289        /**
290         * Register a type serializer for a given type.
291         *
292         * <p>Serializers registered will match all subclasses of the provided
293         * type, as well as unwrapped primitive equivalents of the type.
294         *
295         * @param type the type to accept
296         * @param serializer the serializer that will be serialized with
297         * @param <T> the type to generify around
298         * @return this builder
299         * @since 4.0.0
300         */
301        public <T> Builder register(final TypeToken<T> type, final TypeSerializer<? super T> serializer) {
302            return this.register0(type.getType(), serializer);
303        }
304
305        /**
306         * Register a type serializer for a given type.
307         *
308         * <p>Serializers registered will match all subclasses of the provided
309         * type, as well as unboxed primitive equivalents of the type.
310         *
311         * @param type the type to accept
312         * @param serializer the serializer that will be serialized with
313         * @param <T> the type to generify around
314         * @return this builder
315         * @since 4.0.0
316         */
317        public <T> Builder register(final Class<T> type, final TypeSerializer<? super T> serializer) {
318            return this.register0(type, serializer);
319        }
320
321        /**
322         * Register a type serializer matching against a given predicate.
323         *
324         * @param test the predicate to match types against
325         * @param serializer the serializer to serialize matching types with
326         * @param <T> the type parameter
327         * @return this builder
328         * @since 4.0.0
329         */
330        public <T> Builder register(final Predicate<Type> test, final TypeSerializer<? super T> serializer) {
331            requireNonNull(test, "test");
332            requireNonNull(serializer, "serializer");
333            this.serializers.add(new TypeRegistration(test, serializer));
334            return this;
335        }
336
337        /**
338         * Register a scalar serializer with its own attached type token.
339         *
340         * <p>Serializers registered will match all subclasses of the provided
341         * type, as well as unboxed primitive equivalents of the type.</p>
342         *
343         * @param serializer serializer to register
344         * @param <T> value type
345         * @return this builder
346         * @since 4.0.0
347         */
348        public <T> Builder register(final ScalarSerializer<T> serializer) {
349            requireNonNull(serializer, "serializer");
350            return this.register(serializer.type(), serializer);
351        }
352
353        /**
354         * Register a type serializer matching against a given predicate,
355         * with type annotation information.
356         *
357         * @param test the predicate to match annotated types against
358         * @param serializer the serializer to serialize matching types with
359         * @param <T> the type parameter
360         * @return this builder
361         * @see TypeSerializer.Annotated
362         * @since 4.2.0
363         */
364        public <T> Builder registerAnnotated(final Predicate<AnnotatedType> test, final TypeSerializer<? super T> serializer) {
365            requireNonNull(test, "test");
366            requireNonNull(serializer, "serializer");
367            this.serializers.add(new AnnotatedTypeRegistration(test, serializer));
368            return this;
369        }
370
371        private Builder register0(final Type type, final TypeSerializer<?> serializer) {
372            requireNonNull(type, "type");
373            requireNonNull(serializer, "serializer");
374            this.serializers.add(new TypeRegistration(test -> {
375                // Test direct type
376                if (GenericTypeReflector.isSuperType(type, test)) {
377                    return true;
378                }
379
380                // And upper bounds
381                if (test instanceof WildcardType) {
382                    final Type[] upperBounds = ((WildcardType) test).getUpperBounds();
383                    if (upperBounds.length == 1) {
384                        return isSuperType(type, upperBounds[0]);
385                    }
386                }
387                return false;
388            }, serializer));
389            return this;
390        }
391
392        /**
393         * Register an <em>exact</em> type serializer for a given type.
394         *
395         * <p>Serializers will only match exact object types. For example, a
396         * serializer registered for {@code List<String>} would not match when
397         * {@code ArrayList<String>} is queried.</p>
398         *
399         * @param type the type to accept
400         * @param serializer the serializer that will be serialized with
401         * @param <T> the type to generify around
402         * @return this builder
403         * @since 4.0.0
404         */
405        public <T> Builder registerExact(final TypeToken<T> type, final TypeSerializer<? super T> serializer) {
406            return this.registerExact0(type.getType(), serializer);
407        }
408
409        /**
410         * Register an <em>exact</em> type serializer for a given type.
411         *
412         * <p>Serializers will only match exact object types. For example, a
413         * serializer registered for {@code List<String>} would not match when
414         * {@code ArrayList<String>} is queried.</p>
415         *
416         * @param type the type to accept
417         * @param serializer the serializer that will be serialized with
418         * @param <T> the type to generify around
419         * @return this builder
420         * @since 4.0.0
421         */
422        public <T> Builder registerExact(final Class<T> type, final TypeSerializer<? super T> serializer) {
423            return this.registerExact0(type, serializer);
424        }
425
426        /**
427         * Register a scalar serializer with its own attached type token.
428         *
429         * <p>Serializers will only match exact object types. For example, a
430         * serializer registered for {@code List<String>} would not match when
431         * {@code ArrayList<String>} is queried.</p>
432         *
433         * @param serializer serializer to register
434         * @param <T> value type
435         * @return this builder
436         * @since 4.0.0
437         */
438        public <T> Builder registerExact(final ScalarSerializer<T> serializer) {
439            requireNonNull(serializer, "serializer");
440            return this.registerExact(serializer.type(), serializer);
441        }
442
443        private Builder registerExact0(final Type type, final TypeSerializer<?> serializer) {
444            requireNonNull(type, "type");
445            requireNonNull(serializer, "serializer");
446            this.serializers.add(new TypeRegistration(test -> test.equals(type), serializer));
447            return this;
448        }
449
450        /**
451         * Register all serializers from {@code other} into this collection.
452         *
453         * <p>Creating a child collection should be preferred, but when merging
454         * multiple sets of serializers together, directly adding other
455         * collections may be the best choice.</p>
456         *
457         * @param other source collection
458         * @return this builder
459         * @since 4.0.0
460         */
461        public Builder registerAll(final TypeSerializerCollection other) {
462            this.serializers.addAll(requireNonNull(other, "other").serializers);
463            return this;
464        }
465
466        /**
467         * Register a customized object mapper to handle
468         * {@link ConfigSerializable}-annotated objects.
469         *
470         * @param factory factory to retrieve object mappers from
471         * @return this builder
472         * @since 4.0.0
473         */
474        public Builder registerAnnotatedObjects(final ObjectMapper.Factory factory) {
475            return this.register(Builder::isAnnotatedTarget, factory.asTypeSerializer());
476        }
477
478        /**
479         * A predicate to restrict the type serializer created by
480         * {@link ObjectMapper.Factory#asTypeSerializer()} to annotated types.
481         *
482         * @return whether a type is annotated with {@link ConfigSerializable}
483         * @since 4.0.0
484         */
485        static boolean isAnnotatedTarget(final Type type) {
486            return GenericTypeReflector.annotate(type).isAnnotationPresent(ConfigSerializable.class);
487        }
488
489        /**
490         * Create a new type serializer collection.
491         *
492         * @return a newly created collection
493         * @since 4.0.0
494         */
495        public TypeSerializerCollection build() {
496            return new TypeSerializerCollection(this.parent, this.serializers);
497        }
498    }

500    interface RegisteredSerializer {
501
502        boolean matches(Type test);
503
504        boolean matches(AnnotatedType annotated);
505
506        TypeSerializer<?> serializer();
507
508    }
509
510    static final class TypeRegistration implements RegisteredSerializer {
511
512        private final Predicate<Type> predicate;
513        private final TypeSerializer<?> serializer;
514
515        TypeRegistration(final Predicate<Type> predicate, final TypeSerializer<?> serializer) {
516            this.predicate = predicate;
517            this.serializer = serializer;
518        }
519
520        @Override
521        public boolean matches(final Type test) {
522            return this.predicate.test(test);
523        }
524
525        @Override
526        public boolean matches(final AnnotatedType annotated) {
527            return this.predicate.test(annotated.getType());
528        }
529
530        @Override
531        public TypeSerializer<?> serializer() {
532            return this.serializer;
533        }
534
535    }
536
537    static final class AnnotatedTypeRegistration implements RegisteredSerializer {
538
539        private final Predicate<AnnotatedType> predicate;
540        private final TypeSerializer<?> serializer;
541
542        AnnotatedTypeRegistration(final Predicate<AnnotatedType> predicate, final TypeSerializer<?> serializer) {
543            this.predicate = predicate;
544            this.serializer = serializer;
545        }
546
547        @Override
548        public boolean matches(final Type test) {
549            return this.predicate.test(GenericTypeReflector.annotate(test));
550        }
551
552        @Override
553        public boolean matches(final AnnotatedType annotated) {
554            return this.predicate.test(annotated);
555        }
556
557        @Override
558        public TypeSerializer<?> serializer() {
559            return this.serializer;
560        }
561
562    }
