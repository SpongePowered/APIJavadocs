033/**
034 * Intermediate node type to reduce need for casting.
035 *
036 * <p>Any methods that return {@link ConfigurationNode} in
037 * {@link ConfigurationNode} should be overridden to return the {@code N}
038 * self-type instead.</p>
039 *
040 * @param <N> self type
041 * @since 4.0.0
042 */
043public interface ScopedConfigurationNode<N extends ScopedConfigurationNode<N>> extends ConfigurationNode {
044

046     * Get a correctly typed instance of this node.
047     *
048     * @return the node type
049     * @since 4.0.0
050     */

101    /**
102     * {@inheritDoc}
103     */
104    @Override
105    @SuppressWarnings({"unchecked", "rawtypes"}) // for TypeSerializer.serialize
106    default N set(final Type type, final @Nullable Object value) throws SerializationException {
107        if (value == null) {
108            return this.set(null);
109        }
110        final Type boxed = Types.box(type);
111        final Class<?> erasedType = GenericTypeReflector.erase(boxed);
112        if (!erasedType.isInstance(value)) {
113            throw new SerializationException(this, type, "Got a value of unexpected type "
114                + value.getClass().getName() + ", when the value should be an instance of " + erasedType.getSimpleName());
115        }
116
117        final @Nullable TypeSerializer<?> serial = this.options().serializers().get(boxed);
118        if (serial != null) {
119            try {
120                ((TypeSerializer) serial).serialize(boxed, value, this.self());
121            } catch (final SerializationException ex) {
122                ex.initPath(this::path);
123                ex.initType(type);
124            }
125        } else if (this.options().acceptsType(value.getClass())) {
126            this.raw(value); // Just write if no applicable serializer exists?
127        } else {
128            throw new SerializationException(this, type, "No serializer available for type " + type);
129        }
130        return this.self();
131    }
132
133    /**
134     * {@inheritDoc}
135     */
136    @Override
137    @SuppressWarnings({"unchecked", "rawtypes"}) // for TypeSerializer.serialize
138    default N set(final AnnotatedType type, final @Nullable Object value) throws SerializationException {
139        if (value == null) {
140            return this.set(null);
141        }
142        final Class<?> erasedType = GenericTypeReflector.erase(type.getType());
143        if (!erasedType.isInstance(value)) {
144            throw new SerializationException(this, type, "Got a value of unexpected type "
145                + value.getClass().getName() + ", when the value should be an instance of " + erasedType.getSimpleName());
146        }
147
148        final @Nullable TypeSerializer<?> serial = this.options().serializers().get(type);
149        if (serial != null) {
150            try {
151                ((TypeSerializer) serial).serialize(type, value, this);
152            } catch (final SerializationException ex) {
153                ex.initPath(this::path);
154                ex.initType(type);
155            }
156        } else if (this.options().acceptsType(value.getClass())) {
157            this.raw(value); // Just write if no applicable serializer exists?
158        } else {
159            throw new SerializationException(this, type, "No serializer available for type " + type);
160        }
161        return this.self();
162    }
163
164    @Override
165    default <V> N set(final Class<V> type, final @Nullable V value) throws SerializationException {
166        return this.set((Type) type, value);
167    }
168
169    @Override
170    default <V> N set(final TypeToken<V> type, final @Nullable V value) throws SerializationException {
171        return this.set(type.getType(), value);
172    }

174    @Override
175    default <V> N setList(final Class<V> elementType, final @Nullable List<V> items) throws SerializationException {

177        return this.self();
178    }
179

184    }

228        return (Collector) ConfigurationNode.super.toListCollector(valueType);
229    }
230
231    /**
232     * {@inheritDoc}
233     */
234    @SuppressWarnings({"unchecked", "rawtypes"})
235    @Override
236    default <V> Collector<V, N, N> toListCollector(final Class<V> valueType) {
237        return (Collector) ConfigurationNode.super.toListCollector(valueType);
238    }
239
240    /**
241     * Execute an action on this node. This allows performing multiple
242     * operations on a single node without having to clutter up the surrounding
243     * scope.
244     *
245     * @param <E> thrown type
246     * @param action the action to perform on this node
247     * @return this node
248     * @throws E when thrown by callback {@code action}
249     * @since 4.0.0
250     */
251    default <E extends Exception> N act(final CheckedConsumer<? super N, E> action) throws E {
252        action.accept(this.self());
253        return this.self();
254    }
